  <script>
    // PRODUCTS: change / add items here
    const PRODUCTS = [
      {id:1,title:'Handmade Scarf',price:450,image:svgImg('Scarf')},
      {id:2,title:'Cushion Cover',price:650,image:svgImg('Cushion')},
      {id:3,title:'Ceramic Mug',price:320,image:svgImg('Mug')}
    ];

    // Simple in-memory cart
    const cart = {};

    function svgImg(label){
      const svg = `<svg xmlns='http://www.w3.org/2000/svg' width='300' height='160' viewBox='0 0 300 160'><rect width='100%' height='100%' fill='#eef2ff'/><text x='50%' y='50%' dominant-baseline='middle' text-anchor='middle' font-size='20' fill='#0369a1' font-family='Arial'>${label}</text></svg>`;
      return 'data:image/svg+xml;utf8,'+encodeURIComponent(svg);
    }

    // Render products
    function renderProducts(){
      const root = document.getElementById('products');
      root.innerHTML = '';
      for(const p of PRODUCTS){
        const el = document.createElement('div'); el.className='card';
        el.innerHTML = `
          <div class='product-img'><img src='${p.image}' alt='${p.title}' style='max-width:100%;max-height:100%;border-radius:6px'/></div>
          <div class='product-title'>${p.title}</div>
          <div class='small'>Short product description goes here.</div>
          <div style='display:flex;justify-content:space-between;align-items:center;margin-top:10px'>
            <div class='product-price'>৳${p.price}</div>
            <button onclick='addToCart(${p.id})'>Add</button>
          </div>
        `;
        root.appendChild(el);
      }
    }

    function addToCart(id){
      const prod = PRODUCTS.find(x=>x.id===id);
      if(!prod) return;
      if(!cart[id]) cart[id]={...prod,qty:0};
      cart[id].qty++;
      showCart();
    }

    function showCart(){
      const cartBox = document.getElementById('cartBox');
      const items = document.getElementById('cartItems');
      const count = document.getElementById('cartCount');
      const totalEl = document.getElementById('cartTotal');
      const keys = Object.keys(cart);
      if(keys.length===0){ cartBox.hidden=true; return; }
      cartBox.hidden=false;
      items.innerHTML='';
      let total=0; let itemCount=0;
      for(const k of keys){
        const it = cart[k];
        itemCount += it.qty;
        total += it.qty * it.price;
        const row = document.createElement('div'); row.className='cart-row';
        row.innerHTML = `<div style='flex:1'><div style='font-weight:600'>${it.title}</div><div class='small'>৳${it.price} x ${it.qty} = ৳${it.price*it.qty}</div></div><div style='display:flex;flex-direction:column;align-items:flex-end'><div style='display:flex;gap:6px'><button onclick='dec(${it.id})'>-</button><div class='qty'>${it.qty}</div><button onclick='inc(${it.id})'>+</button></div><button style='margin-top:6px;background:#f8d7da;color:#7f1d1d' onclick='remove(${it.id})'>Remove</button></div>`;
        items.appendChild(row);
      }
      count.textContent = itemCount;
      totalEl.textContent = '৳' + total;
    }

    function inc(id){ cart[id].qty++; showCart(); }
    function dec(id){ cart[id].qty--; if(cart[id].qty<=0) delete cart[id]; showCart(); }
    function remove(id){ delete cart[id]; showCart(); }

    // Checkout
    document.getElementById('checkoutBtn').addEventListener('click',()=>{
      const keys = Object.keys(cart);
      if(keys.length===0){ alert('আপনার কার্ট খালি।'); return; }
      const sum = keys.reduce((s,k)=>s + cart[k].price*cart[k].qty,0);
      document.getElementById('checkoutSummary').innerHTML = `<div class='small'>${keys.length} প্রোডাক্ট — মোট ৳${sum}</div>`;
      document.getElementById('checkoutModal').style.display='flex';
    });
    document.getElementById('cancelCheckout').addEventListener('click',()=>{document.getElementById('checkoutModal').style.display='none'});

    document.getElementById('checkoutForm').addEventListener('submit',e=>{
      e.preventDefault();
      // Demo: show fake success
      const name = e.target.name.value;
      document.getElementById('checkoutModal').style.display='none';
      alert(`ধন্যবাদ ${name}! আপনার অর্ডার নেওয়া হয়েছে (ডেমো) — কার্ট রিসেট হয়েছে।`);
      // reset cart
      for(const k of Object.keys(cart)) delete cart[k];
      showCart();
    });

    // init
    renderProducts();
    showCart();

  </script>
